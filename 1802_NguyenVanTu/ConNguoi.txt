public class ConNguoi{

	private int tuoi;
	private String gioiTinh;
	private int cmnd;

	}

	}
	public void setTuoi(int tuoi){
	this.tuoi = tuoi;
	}
	public String getGioiTinh(){
	return gioiTinh;
	}
	public void setGioiTinh(String gioiTinh){
	this.gioiTinh = gioiTinh;
	}

	return cmnd;
	}

	this.cmnd = cmnd;
	}

	public String toString(){

}
